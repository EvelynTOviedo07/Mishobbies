    <script>
        // --- LÓGICA DEL REPRODUCTOR DE AUDIO ---
        const miAudio = document.getElementById('bg-music');
        const btnPlay = document.getElementById('btn-play');
        const btnPause = document.getElementById('btn-pause');

        // Acción al presionar Play
        btnPlay.addEventListener('click', () => {
            miAudio.play();
        });

        // Acción al presionar Pause
        btnPause.addEventListener('click', () => {
            miAudio.pause();
        });


        // --- LÓGICA DE DESPLAZAMIENTO SUAVE INTERACTIVO ---
        // Buscamos todos los enlaces que tengan un "href" apuntando a un id (#)
        document.querySelectorAll('.nav-link').forEach(enlace => {
            enlace.addEventListener('click', function(e) {
                e.preventDefault(); // Evita el salto brusco instantáneo nativo
                
                // Obtenemos la sección de destino mediante el atributo href
                const targetId = this.getAttribute('href');
                const targetSection = document.querySelector(targetId);
                
                // Calculamos la posición restando la altura de la barra de navegación para que no tape el título
                const navbarHeight = document.querySelector('.navbar').offsetHeight;
                const targetPosition = targetSection.offsetTop - navbarHeight;

                // Hacemos el desplazamiento scroll personalizado por JS
                window.scrollTo({
                    top: targetPosition,
                    behavior: 'smooth'
                });
            });
        });
    </script>
