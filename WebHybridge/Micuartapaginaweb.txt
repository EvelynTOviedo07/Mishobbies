    <script>
        // --- 1. SCRIPT DE LA PLAYLIST DE MÚSICA ---
        const audio = document.getElementById('bg-music');
        const playBtn = document.getElementById('btn-play');
        const pauseBtn = document.getElementById('btn-pause');
        const playlistSelect = document.getElementById('playlist-select');

        // Al cambiar de opción en el menú desplegable, cambia la canción
        playlistSelect.addEventListener('change', function() {
            const estabaReproduciendo = !audio.paused;
            audio.src = this.value; // Asigna el archivo mp3 seleccionado
            if (estabaReproduciendo) {
                audio.play();
            }
        });

        playBtn.addEventListener('click', () => audio.play());
        pauseBtn.addEventListener('click', () => audio.pause());


        // --- 2. DESPLAZAMIENTO SUAVE ---
        document.querySelectorAll('.nav-link').forEach(enlace => {
            enlace.addEventListener('click', function(e) {
                e.preventDefault();
                const targetId = this.getAttribute('href');
                const targetSection = document.querySelector(targetId);
                const navbarHeight = document.querySelector('.navbar').offsetHeight;
                const targetPosition = targetSection.offsetTop - navbarHeight;

                window.scrollTo({
                    top: targetPosition,
                    behavior: 'smooth'
                });
            });
        });


        // --- 3. SISTEMA DE COMENTARIOS (BLOG/LOCALSTORAGE) ---
        const commentForm = document.getElementById('comment-form');
        const commentsContainer = document.getElementById('comments-container');

        // Cargar mensajes guardados al abrir la página
        document.addEventListener('DOMContentLoaded', loadComments);

        commentForm.addEventListener('submit', function(e) {
            e.preventDefault(); // Evita que se recargue la página

            const name = document.getElementById('author-name').value;
            const text = document.getElementById('comment-text').value;
            const date = new Date().toLocaleDateString() + " " + new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});

            const newComment = { name, text, date };

            // Obtener los guardados, meter el nuevo y volver a guardar
            let comments = JSON.parse(localStorage.getItem('blogComments')) || [];
            comments.unshift(newComment); // Añade al inicio de la lista
            localStorage.setItem('blogComments', JSON.stringify(comments));

            // Limpiar el formulario e imprimir
            document.getElementById('comment-text').value = '';
            displayComments();
        });

        function loadComments() {
            displayComments();
        }

        function displayComments() {
            let comments = JSON.parse(localStorage.getItem('blogComments')) || [];
            commentsContainer.innerHTML = ''; // Limpiar contenedor

            if(comments.length === 0) {
                commentsContainer.innerHTML = '<p style="font-size:0.9rem; color:#777; text-align:center;">No hay mensajes todavía. ¡Sé el primero!</p>';
                return;
            }

            comments.forEach(c => {
                const commentElement = document.createElement('div');
                commentElement.className = 'single-comment';
                commentElement.innerHTML = `
                    <div style="display:flex; justify-content:space-between; margin-bottom:5px;">
                        <span class="comment-author">👤 ${escapeHTML(c.name)}</span>
                        <span style="font-size:0.75rem; color:#777;">${c.date}</span>
                    </div>
                    <p style="margin:0; font-size:0.9rem; color:#e0e0e6;">${escapeHTML(c.text)}</p>
                `;
                commentsContainer.appendChild(commentElement);
            });
        }

        // Función de seguridad para evitar que rompan el código metiendo etiquetas HTML en los campos
        function escapeHTML(str) {
            return str.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;");
        }
    </script>
